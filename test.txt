    x real $100
    y integer $100
    z text "hello"
endVariable


state welcome
    speak "Hello, world " 
    speak "Hola, mundo!" "Bonjour, monde!"
    speak "Input balance to check your account's balance"
    switch
        case "hello" 
            speak "hello"
            goto welcome
        case "balance"
            speak "Your balance is " x
            goto welcome
        case "bye" 
            speak "bye"
        case "exit" 
            goto goodbye
        default 
            speak "default"
    endSwitch
    timeout $10 
        speak "You have been idle for 10 seconds. Restarting service ..."
        goto welcome
    endTimeout
endState

    speak "Goodbye, world!"
    switch 
        case "goodbye"
            exit
        default
            speak "not quite understood"
    endSwitch

            goto welcome
    endTimeout
